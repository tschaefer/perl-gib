=head2 C<sub to_markdown($self); #59>

Return Markdown.

 my $doc = Perl::Gib::Markdown->new( { file => 'lib/Perl/Gib/Usage.md' } );
 
 my $markdown = $doc->to_markdown();
 my @lines    = split /\n/, $markdown;
 is( $lines[0], '# perlgib', 'Markdown documentation' );
